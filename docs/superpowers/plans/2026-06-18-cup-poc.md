**Goal:** Parameter DAG 기반 서브그래프로 파트 간 접합 무결성과 파트 독립 조합을 증명하는 FastAPI + Cascade Studio POC

```python
from dataclasses import dataclass, field
from typing import Callable, Optional


@dataclass
class ParamNode:
    name: str
    value: float
    subgraph: str  # "body", "handle", "joint"
    is_user_facing: bool
    formula: Optional[Callable[["DAG"], float]] = None
    dependencies: list[str] = field(default_factory=list)


class DAG:
    def __init__(self) -> None:
        self._nodes: dict[str, ParamNode] = {}

    def add_node(self, node: ParamNode) -> None:
        self._nodes[node.name] = node

    def get(self, name: str) -> float:
        return self._nodes[name].value

    def set(self, name: str, value: float) -> None:
        self._nodes[name].value = value

    def get_all(self) -> dict[str, float]:
        return {name: node.value for name, node in self._nodes.items()}

    def get_user_facing(self) -> dict[str, float]:
        return {name: node.value for name, node in self._nodes.items() if node.is_user_facing}
```

```python
from collections import defaultdict, deque
from app.graph.model import DAG


def _dependents_map(dag: DAG) -> dict[str, list[str]]:
    """노드 이름 → 이 노드에 의존하는 노드 이름 목록"""
    result: dict[str, list[str]] = defaultdict(list)
    for node in dag._nodes.values():
        for dep in node.dependencies:
            result[dep].append(node.name)
    return result


def _topo_sort_affected(dag: DAG, changed_names: list[str]) -> list[str]:
    """changed_names 로부터 영향받는 파생 노드를 위상 정렬 순서로 반환."""
    dependents = _dependents_map(dag)

    # BFS로 영향받는 노드 집합 수집
    visited: set[str] = set()
    queue: deque[str] = deque()
    for name in changed_names:
        for dep in dependents.get(name, []):
            if dep not in visited:
                visited.add(dep)
                queue.append(dep)
    while queue:
        current = queue.popleft()
        for dep in dependents.get(current, []):
            if dep not in visited:
                visited.add(dep)
                queue.append(dep)

    # 위상 정렬 (의존성 만족된 노드부터 처리)
    result: list[str] = []
    processed: set[str] = set(dag._nodes.keys()) - visited  # 영향 없는 노드는 처리된 것으로
    remaining = list(visited)

    while remaining:
        progress = False
        for name in remaining[:]:
            node = dag._nodes[name]
            if all(dep in processed for dep in node.dependencies):
                result.append(name)
                processed.add(name)
                remaining.remove(name)
                progress = True
        if not progress:
            break  # 순환 의존이 있는 경우 (POC 범위에서는 발생하지 않음)

    return result


def initialize(dag: DAG) -> None:
    """모든 파생 노드를 처음부터 위상 정렬 순서로 계산."""
    user_nodes = {n for n, node in dag._nodes.items() if node.formula is None}
    remaining = [n for n in dag._nodes if dag._nodes[n].formula is not None]
    processed = set(user_nodes)

    while remaining:
        progress = False
        for name in remaining[:]:
            node = dag._nodes[name]
            if all(dep in processed for dep in node.dependencies):
                dag.set(name, node.formula(dag))
                processed.add(name)
                remaining.remove(name)
                progress = True
        if not progress:
            break


def recompute(dag: DAG, changed_names: list[str]) -> tuple[list[str], list[str]]:
    """
    변경된 노드 목록을 받아 영향받는 파생 노드를 순서대로 재계산.
    반환: (changed_names, recomputed_names)
    """
    order = _topo_sort_affected(dag, changed_names)
    recomputed: list[str] = []
    for name in order:
        node = dag._nodes[name]
        if node.formula is not None:
            dag.set(name, node.formula(dag))
            recomputed.append(name)
    return changed_names, recomputed
```

```python
    dag.add_node(ParamNode("lower_attach_z_ratio", p["lower_attach_z_ratio"], "handle", True))
    dag.add_node(ParamNode("upper_attach_z_ratio", p["upper_attach_z_ratio"], "handle", True))
    dag.add_node(ParamNode("outward_depth_ratio", p["outward_depth_ratio"], "handle", True))


def _build_tapered_bspline_joint(dag: DAG) -> None:
    def outer_r_at_z(d: DAG, z: float) -> float:
        h = d.get("height")
        return d.get("outer_bottom_r") + (d.get("outer_top_r") - d.get("outer_bottom_r")) * (z / h)

    dag.add_node(ParamNode(
        "lower_attach_z", 0.0, "joint", False,
        formula=lambda d: d.get("height") * d.get("lower_attach_z_ratio"),
        dependencies=["height", "lower_attach_z_ratio"],
    ))
    dag.add_node(ParamNode(
        "upper_attach_z", 0.0, "joint", False,
        formula=lambda d: d.get("height") * d.get("upper_attach_z_ratio"),
        dependencies=["height", "upper_attach_z_ratio"],
    ))
    dag.add_node(ParamNode(
        "handle_span_z", 0.0, "joint", False,
        formula=lambda d: d.get("upper_attach_z") - d.get("lower_attach_z"),
        dependencies=["upper_attach_z", "lower_attach_z"],
    ))
    dag.add_node(ParamNode(
        "lower_outer_r", 0.0, "joint", False,
        formula=lambda d: outer_r_at_z(d, d.get("lower_attach_z")),
        dependencies=["outer_bottom_r", "outer_top_r", "height", "lower_attach_z"],
    ))
    dag.add_node(ParamNode(
        "upper_outer_r", 0.0, "joint", False,
        formula=lambda d: outer_r_at_z(d, d.get("upper_attach_z")),
        dependencies=["outer_bottom_r", "outer_top_r", "height", "upper_attach_z"],
    ))
    dag.add_node(ParamNode(
        "penetration_depth", 0.0, "joint", False,
        formula=lambda d: d.get("thickness") * 0.75,
        dependencies=["thickness"],
    ))
    dag.add_node(ParamNode(
        "lower_attach_x", 0.0, "joint", False,
        formula=lambda d: -d.get("lower_outer_r") + d.get("penetration_depth"),
        dependencies=["lower_outer_r", "penetration_depth"],
    ))
    dag.add_node(ParamNode(
        "upper_attach_x", 0.0, "joint", False,
        formula=lambda d: -d.get("upper_outer_r") + d.get("penetration_depth"),
        dependencies=["upper_outer_r", "penetration_depth"],
    ))
    dag.add_node(ParamNode(
        "handle_mid_z", 0.0, "joint", False,
        formula=lambda d: (d.get("lower_attach_z") + d.get("upper_attach_z")) / 2,
        dependencies=["lower_attach_z", "upper_attach_z"],
    ))
    dag.add_node(ParamNode(
        "handle_mid_outer_r", 0.0, "joint", False,
        formula=lambda d: outer_r_at_z(d, d.get("handle_mid_z")),
        dependencies=["outer_bottom_r", "outer_top_r", "height", "handle_mid_z"],
    ))
    dag.add_node(ParamNode(
        "handle_outward_depth", 0.0, "joint", False,
        formula=lambda d: d.get("outer_top_r") * d.get("outward_depth_ratio"),
        dependencies=["outer_top_r", "outward_depth_ratio"],
    ))


def _build_cylinder_body(dag: DAG, params: dict) -> None:
    p = {**CYLINDER_BODY_DEFAULTS, **params}
    dag.add_node(ParamNode("cylinder_height", p["cylinder_height"], "body", True))
    dag.add_node(ParamNode("cylinder_radius", p["cylinder_radius"], "body", True))
    dag.add_node(ParamNode("cylinder_thickness", p["cylinder_thickness"], "body", True))
    dag.add_node(ParamNode(
        "cylinder_inner_radius", 0.0, "body", False,
        formula=lambda d: d.get("cylinder_radius") - d.get("cylinder_thickness"),
        dependencies=["cylinder_radius", "cylinder_thickness"],
    ))
    dag.add_node(ParamNode(
        "cylinder_bottom_thickness", 0.0, "body", False,
        formula=lambda d: d.get("cylinder_thickness"),
        dependencies=["cylinder_thickness"],
    ))


def _build_ring_handle(dag: DAG, params: dict) -> None:
    p = {**RING_HANDLE_DEFAULTS, **params}
    dag.add_node(ParamNode("ring_outer_r", p["ring_outer_r"], "handle", True))
    dag.add_node(ParamNode("ring_inner_r", p["ring_inner_r"], "handle", True))
    dag.add_node(ParamNode("ring_width", p["ring_width"], "handle", True))
    dag.add_node(ParamNode("ring_attach_z", p["ring_attach_z"], "handle", True))


def _build_cylinder_ring_joint(dag: DAG) -> None:
    dag.add_node(ParamNode(
        "ring_attach_x", 0.0, "joint", False,
        formula=lambda d: d.get("cylinder_radius") + d.get("ring_outer_r") - d.get("cylinder_thickness") * 0.75,
        dependencies=["cylinder_radius", "ring_outer_r", "cylinder_thickness"],
    ))


# ── 공개 API ─────────────────────────────────────────────────

def build_dag(body_template: str, handle_template: str | None, params: dict | None = None) -> DAG:
    """
    body_template: "tapered" | "cylinder"
    handle_template: "bspline" | "ring" | None
    params: 사용자 입력값 오버라이드 (user-facing 파라미터만 유효)
    """
    dag = DAG()
    p = params or {}

    if body_template == "tapered":
        _build_tapered_body(dag, p)
    elif body_template == "cylinder":
        _build_cylinder_body(dag, p)
    else:
        raise ValueError(f"Unknown body_template: {body_template}")

    if handle_template == "bspline":
        _build_bspline_handle(dag, p)
        _build_tapered_bspline_joint(dag)
    elif handle_template == "ring":
        _build_ring_handle(dag, p)
        _build_cylinder_ring_joint(dag)
    elif handle_template is not None:
        raise ValueError(f"Unknown handle_template: {handle_template}")

    return dag
```

```python
from pathlib import Path
from jinja2 import Environment, FileSystemLoader
from app.graph.model import DAG

_FEATURES_DIR = Path(__file__).parent.parent / "features"
_ENV = Environment(loader=FileSystemLoader(str(_FEATURES_DIR)))


def render(body_template: str, handle_template: str | None, dag: DAG) -> str:
    """DAG 파라미터 값으로 Jinja2 JS 템플릿을 렌더링하고 조합된 코드를 반환."""
    params = dag.get_all()
    parts: list[str] = []

    body_tmpl = _ENV.get_template(f"body/{body_template}_body.js.j2")
    parts.append(body_tmpl.render(**params))

    if handle_template:
        handle_tmpl = _ENV.get_template(f"handle/{handle_template}_handle.js.j2")
        parts.append(handle_tmpl.render(**params))
        parts.append("\n// === UNION ===\nlet cup = Union([cupBody, cupHandle]);")
    else:
        parts.append("\n// === RESULT ===\ncupBody;")

    return "\n\n".join(parts)
```

```python
- BSpline handle: section_width, section_height, lower_attach_z_ratio, upper_attach_z_ratio, outward_depth_ratio

Return ONLY a JSON array of {"param": "param_name", "value": number} objects.
Example: [{"param": "height", "value": 90}]
Match the closest parameter name if the user uses natural language."""


def parse_command(command: str, current_params: dict) -> list[dict]:
    """자연어 명령을 [{param, value}] 리스트로 파싱."""
    response = client.messages.create(
        model="claude-haiku-4-5-20251001",
        max_tokens=256,
        system=_SYSTEM_PROMPT,
        messages=[
            {
                "role": "user",
                "content": f"Current params: {json.dumps(current_params)}\n\nCommand: {command}",
            }
        ],
    )
    text = response.content[0].text.strip()
    start = text.find("[")
    end = text.rfind("]") + 1
    return json.loads(text[start:end])
```

```python
from typing import Optional
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from app.graph.cup_graph import build_dag
from app.graph.solver import initialize, recompute
from app.codegen.renderer import render
from app.llm.parser import parse_command

router = APIRouter(prefix="/api")


class ModelState(BaseModel):
    body_template: str
    handle_template: Optional[str] = None
    params: dict = {}


class ModelRequest(BaseModel):
    body_template: str
    handle_template: Optional[str] = None
    params: dict = {}


class GenerateRequest(BaseModel):
    command: str
    state: Optional[ModelState] = None


class ModelResponse(BaseModel):
    code: str
    state: ModelState


class GenerateResponse(BaseModel):
    code: str
    state: ModelState
    diff: dict


def _build_and_init(body_template: str, handle_template: Optional[str], params: dict):
    dag = build_dag(body_template, handle_template, params)
    initialize(dag)
    return dag


@router.post("/model", response_model=ModelResponse)
def create_model(req: ModelRequest):
    try:
        dag = _build_and_init(req.body_template, req.handle_template, req.params)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    code = render(req.body_template, req.handle_template, dag)
    return ModelResponse(
        code=code,
        state=ModelState(
            body_template=req.body_template,
            handle_template=req.handle_template,
            params=dag.get_all(),
        ),
    )


@router.post("/generate", response_model=GenerateResponse)
def generate(req: GenerateRequest):
    state = req.state or ModelState(body_template="tapered", handle_template="bspline")
    try:
        dag = _build_and_init(state.body_template, state.handle_template, state.params)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))

    user_params = dag.get_user_facing()
    changes = parse_command(req.command, user_params)

    changed_names: list[str] = []
    for change in changes:
        param = change.get("param")
        value = change.get("value")
        if param and param in dag._nodes and dag._nodes[param].is_user_facing:
            dag.set(param, float(value))
            changed_names.append(param)

    _, recomputed_names = recompute(dag, changed_names)
    code = render(state.body_template, state.handle_template, dag)

    return GenerateResponse(
        code=code,
        state=ModelState(
            body_template=state.body_template,
            handle_template=state.handle_template,
            params=dag.get_all(),
        ),
        diff={"changed": changed_names, "recomputed": recomputed_names},
    )
```
